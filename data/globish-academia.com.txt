  var menuIdMap = {

  function showMenuBox(menuId) {
    if(document.getElementById("topNav").classList.contains("small-screen-expand")) {
      return;
    }
    if(menuIdMap[menuId] != undefined) {
      menuIdMap[menuId].classList.add("show");
    }
    document.getElementById("mainMenu").classList.add("active");
    document.getElementById("rightMenu").classList.add("active");
  }

  function hideMenuBox(menuId) {
    if(document.getElementById("topNav").classList.contains("small-screen-expand")) {
      return;
    }
    if(menuIdMap[menuId] != undefined) {
      menuIdMap[menuId].classList.remove("show");
    }
    document.getElementById("mainMenu").classList.remove("active");
    document.getElementById("rightMenu").classList.remove("active");
  }

  function hideAllMenuBoxExcept(menuIdToExclude) {
    for(var menuId in menuIdMap) {
      if(menuId != menuIdToExclude) {
        menuIdMap[menuId].classList.remove("show-small-screen");
        menuIdMap[menuId].parentElement.classList.remove("active-small-screen");
      }
    }
  }

  function toggleMenuBox(target, menuId) {
    if(menuIdMap[menuId] != undefined) {
      menuIdMap[menuId].classList.toggle("show-small-screen");
      if(menuIdMap[menuId].classList.contains("show-small-screen")) {
        target.classList.add("active-small-screen");
        hideAllMenuBoxExcept(menuId);

        document.getElementById("mainMenu").classList.add("active-small-screen");
        document.getElementById("rightMenu").classList.add("active-small-screen");
      } else {
        target.classList.remove("active-small-screen");

        document.getElementById("mainMenu").classList.remove("active-small-screen");
        document.getElementById("rightMenu").classList.remove("active-small-screen");
      }
    }
  }

  }
