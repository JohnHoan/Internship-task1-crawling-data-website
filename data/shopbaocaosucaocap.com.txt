$('body').on('click', '.AddToCart', function(event) {
  event.preventDefault();
  var pid=$(this).attr('data-id');
  var soluong=$('#qty').val();
  if(!soluong) soluong=1;
  $.ajax ({
    type: "POST",
    url: "ajax/add_giohang.php",
    dataType : 'json',
    data: {pid:pid,soluong:soluong},
    success: function(result) { 
      $('.num_cart').html(result.count);
      $('.soluong_e').html(result.count);
      if(result.html!='') {
        $('.cart_wrapper table tbody').append(result.html);
        $('#total_price_popup').html(result.tonggia);
      }
      else
      {
        $('.cart-'+pid).find('.product_quantity').val(result.soluong);
        $('.cart-'+pid).find('.price_total').html(result.tongtien);
        $('#total_price_popup').html(result.tonggia);
      }
      $('#myModal').modal('show');
    }
  });
});

function updownqty(type)
{
  var soluong=$('.qty-text').val();
  if(type=='up')
  {
    soluong++;
    $('.qty-text').val(soluong);
  }
  if(type=='down')
  {
    if(soluong > 1)
    {
      soluong--;
      $('.qty-text').val(soluong);
    }
  }
}
