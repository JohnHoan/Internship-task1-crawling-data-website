<script type="text/javascript">
	function get_price_detail(){

		var id = $('.id_sp').val();
		var sl = $('#soluong').val();
		var id_mau=$('.active_color').attr('data-id');

		var id_size=$('.active_size').attr('data-id');
		if(typeof(id_mau)  === "undefined"){
			id_mau=0;
		}
		if(typeof(id_size)  === "undefined"){
			id_size=0;
		}
		$.ajax({
			url:'ajax/sum_price.php',
			data:{id:id,sl:sl,id_mau:id_mau,id_size:id_size},
			type:'POST',
			async:false,
			success: function(res){
				$('.sum_price').html(res)
			}
		});

	}

</script>
