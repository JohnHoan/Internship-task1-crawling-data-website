	function addtocart10(act, pid, quantity){
		var color = 0;
		var size = 0;
		if($(".pd-detail__color.active").length > 0) {
			color = $(".pd-detail__color.active").data('id');
		}
		if($(".pd-detail__size.active").length > 0) {
			size = $(".pd-detail__size.active").data('id');
		}
		$.ajax({
			url: 'ajax/cart.php',
			type: 'POST',
			dataType: 'json',
			data: {act:act, pid:pid, quantity:quantity, color:color, size:size},
		})
		.done(function(result) {
			if(act == 'addtocart'){
				$.fancybox.destroy();
				Swal.fire({
					icon: 'success',
					title: 'Thêm sản phẩm vào giỏ hàng thành công',
					showConfirmButton: false,
					timer: 1500
				})
			}
			if(act == 'buynow'){
				window.location = 'gio-hang';
			}
			$('.quantity-order').html(result.quantity);
		})
		.fail(function() {
			console.log("error");
		});
	}
