            function autocomplet()
            {
                var min_length = 0; // min caracters to display the autocomplete
                var keyword = $('#filter-s').val();
                if (keyword.length >= min_length) {
                    $.ajax({
                        url: '/ajax/',
                        type: 'POST',
                        data: "keyword="+keyword+"&task_ajax=task_viewbranch",
                        success:function(data){
                            var result = JSON.parse(data);
                            if(result.result)
                            {
                                $('#list_branch').show();
                                $('#list_branch').html(result.result);
                            }
                            else
                            {
                              $('#list_branch').hide();
                            }
                        }
                    });
                } 
                else 
                {
                    $('#list_branch').hide();
                }
            }
            
            $(window).click(function(){
                $('#list_branch').hide();

<script>
$(window).load(function(){
    countCart();
})    

function countCart()
{
    $.ajax({
        type:"POST",
        url:"/ajax/",
        data: 'task_ajax=task_CountCart',
        success:function(data){
            var result = JSON.parse(data);
            $('.mini-cart-counter').empty().append(result.result);
        }
    });
}

function addQuickCart(id)
{
    $.ajax({
        type:"POST",
        url:'/ajax/',
        data: "task_ajax=task_AddQuickCart&id=" + id,
        success:function(data){
            var result = JSON.parse(data);
            if(result.redirect)
            {
                window.location = result.redirect;
            }
            else
            {
                showDlgMessage("Giỏ hàng", result.result, 'auto', 'auto');
                countCart();
            }
        }
    });
}
</script>
