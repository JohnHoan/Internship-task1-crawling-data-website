        return request("http://saolakids.vn/cart", {'id': productId, 'quantity': quantity, 'color': color}, "#top-cart");
    }

    function removeCartItem(productId) {
        return request("http://saolakids.vn/cart/remove", {'id': productId}, "#top-cart");
    }

    function removeCartItemOnlist(productId) {
        return request("http://saolakids.vn/cart/remove", {'id': productId, 'view': 'frontend.cart._list_page'}, "#detail-cart");
    }

    function updateCartItem(productId, quantity) {
        return request("http://saolakids.vn/cart/update", {'id': productId, 'quantity': quantity}, "#detail-cart");
    }

    function reloadCart() {
        setTimeout(function() {
            request("http://saolakids.vn/cart/reload", {'id': ''}, "#top-cart");
        }, 1200);
    }

            showConfirmButton: false,

        })

    $('body').on('click', '.cart_item .remove',  function(e) {
        productId = $(this).attr('data-product-id');

        e.preventDefault();

        Swal.fire({
            title: 'Xóa sản phẩm khỏi giỏ hàng? Vui lòng xác nhận.',
            showCancelButton: true,
            confirmButtonText: 'Xóa'
        }).then((result) => {
            if (result.isConfirmed) {
                removeCartItemOnlist(productId);
                reloadCart();
            }
        });
        e.stopPropagation();
    });

    // View Cart hover items toggle show hide
    if ($('.cart-view').has('.cart-popup')) {
        $('body').on('click', '.has-cart-data',  function(e) {
            e.preventDefault();
            if ($(this).parent('.cart-view').hasClass('show')) {
                $(this).parent('.cart-view').removeClass('show');
            } else {
                $(this).parent('.cart-view').addClass('show');
            }
            e.stopPropagation();
        });
    }

    // Quantity number changing script
    $('<div class="quantity-nav"><div class="quantity-button quantity-down">-</div><div class="quantity-button quantity-up">+</div></div>').insertAfter('.quantity input');

    $('body').on('click', '.quantity-up',  function(e) {
        e.preventDefault();
        input = $(this).parent().parent().find('input[type="number"]');
        productId = input.attr('data-product-id');
        min = input.attr('min');
        max = input.attr('max');

        var oldValue = parseFloat(input.val());
        if (oldValue >= max) {
            var newVal = oldValue;
        } else {
            var newVal = oldValue + 1;
        }
        input.val(newVal);
        input.trigger("change");

        updateCartItem(productId, newVal);
        reloadCart();
        e.stopPropagation();
    });

    $('body').on('click', '.quantity-down',  function(e) {
        input = $(this).parent().parent().find('input[type="number"]');
        productId = input.attr('data-product-id');
        min = input.attr('min');
        max = input.attr('max');

        var oldValue = parseFloat(input.val());
        if (oldValue <= min) {
            var newVal = oldValue;
        } else {
            var newVal = oldValue - 1;
        }
        input.val(newVal);
        input.trigger("change");

        updateCartItem(productId, newVal);
        reloadCart();
        e.stopPropagation();
    });

    // Click out the cart
    $(document).on('click', function(e) {
        var container = $('.cart-popup');
        if (!container.is(e.target) && container.has(e.target).length === 0) {
            if ($('.cart-view').hasClass('show')) {
                $('.cart-view').removeClass('show');
            }
        }
    });
