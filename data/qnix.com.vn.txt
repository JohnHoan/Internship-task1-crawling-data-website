<script>
var ih = 960;
var stime = 4000; // 1초 = 1000
$("#w_main, .w_slide, .w_slide a").css("height",ih);
$(".w_slide a:first").clone().appendTo(".w_slide");

var num = $(".w_slide a").length;
for (var i=0; i<num; i++) {
	$(".w_slide a").eq(i).css("left",i*100+"%");
	if (i < num-1) {
	$("#w_page").append("<li data-n='"+i+"'><a href='#none'></a></li>");
	};
}

$("#w_page li:first").addClass("select");

var n=0;
function auto() {
	n++;
	if (n==num-1) { 
		$(".w_slide").animate({"left":-n*100+"%"},300, function(){
			$(this).css("left",0);	
		});
		n=0; 
	} else {
		$(".w_slide").animate({"left":-n*100+"%"},300);
	};
	$("#w_page li").removeClass();
	$("#w_page li").eq(n).addClass("select");
};

var timer = setInterval("auto()", stime);

$("#w_page li").click(function(){
	clearInterval(timer);	
	$("#w_page li").removeClass();
	$(this).addClass("select");	
	n=$(this).attr("data-n");
	$(".w_slide").animate({"left":-n*100+"%"},300);
	timer = setInterval("auto()", stime);
});


$("#w_main").hover(
	function(){
		$(".prev").stop().animate({"left":100,"opacity":1},300);
		$(".next").stop().animate({"right":100,"opacity":1},300);
	},
	function(){
		$(".prev").stop().animate({"left":150,"opacity":0},300);
		$(".next").stop().animate({"right":150,"opacity":0},300);
	}
);

$(".next").click(function(){
	clearInterval(timer);
	auto();
	timer = setInterval("auto()", stime);
});

$(".prev").click(function(){
	clearInterval(timer);
	if(n == 0){
		n=num-2;
		$(".w_slide").css({"left":(-n-1)*100+"%"}).animate({"left":-n*100+"%"});
	} else {
		n--;
		$(".w_slide").animate({"left":-n*100+"%"},300);
	}

	$("#w_page li").removeClass();
	$("#w_page li").eq(n).addClass("select");
	timer = setInterval("auto()", stime);
});
</script>
