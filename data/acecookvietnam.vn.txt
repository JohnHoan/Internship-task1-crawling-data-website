		function validateCustomFormnum(number,min,max){
			if (isNaN(number) ) {
				return 1;
			}
			else {
				if(min){
					if(number < min){
						return 3;
					}				
				}
				if(max){
					if(number > max){
						return 2;
					}				
				}
				return 0;
			}
		}
		//date filed validation function
		function validateCustomFordate(input) {
			var status = false;
			if (!input || input.length <= 0) {
			  status = false;
			} 
			else {
			  var result = new Date(input);
			  if (result == 'Invalid Date') {
				status = false;
			  } 
			  else {
				status = true;
			  }
			}
			return status;		
		}
