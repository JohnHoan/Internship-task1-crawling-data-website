    /* search */
    var engine = new Bloodhound({
        remote: {
            url: '/search?keyword=%QUERY%'+'&th=nav',
            wildcard: '%QUERY%'
        },
        datumTokenizer: Bloodhound.tokenizers.whitespace('keyword'),
        queryTokenizer: Bloodhound.tokenizers.whitespace
    });

    $(".search_nav_input:visible").first().typeahead({
        hint: false,
        minLength: 1
    }, [
        {
            // displayKey: 'value',
            source: engine.ttAdapter(),
            name: 'product-title',
            limit: 1000,
            display: function(data) {
                data.url = '/search?keyword='+data.keyword
                return $('.search_nav_input:focus').typeahead('val')
            },
            templates: {
                suggestion: function (data) {
                    ct = '<a href="/search?keyword='+data.keyword+'" class="list-group-item typeahead">'+data.keyword+'</a>';
                    return ct;
                },
            }
        }
    ]);
    $('.search_nav_input:visible').first().bind('typeahead:selected', function(evt, item) {
        document.location.href = item.url
    })
    $('.search_nav_input:visible').first().bind('keypress', function(e) {
        if(e.which == 13) {
            e.preventDefault()
            if($('.search_nav_input:visible').first().nextAll('.tt-menu:first').find('.tt-cursor').length > 0){
                document.location.href = $('.search_nav_input:visible').first().nextAll('.tt-menu:first').find('.tt-cursor').attr('href')
            }else if($(this).val() == ''){
                document.location.href = '/search'
            }else if($(this).val() !== ''){
                document.location.href = '/search?keyword='+$(this).val()
            }
        }
    });
    $('.search_nav_input:visible').first().bind('typeahead:open', function(evt) {
        // for css input
        $('.search_nav_input:visible').first().addClass('open')
        $(this).parents('.nav_search:first').addClass('open')
        // end for css input
        if(evt.target.value == ''){
            getGoiY()
        }
    });
    $('.search_nav_input:visible').first().bind('typeahead:close', function() {
        // for css input
        $('.search_nav_input:visible').first().removeClass('open')
        $(this).parents('.nav_search:first').removeClass('open')
        // end for css input
    });
    function getGoiY(){
        $.ajax({
            url: '/search/get-goi-y',
            type:'get',
            async: false,
            beforeSend: function() {

            },
            success: function(data) {
                ct = ''
                for(i=0;i<data.length;i++){
                    if(data[i].goi_y == 1){
                        ct += '<a class="list-group-item typeahead tt-selectable" href="/search?keyword='+data[i][0]+'">'+data[i][0]+'<img src="/images/icons/hot-search.png"></a>'
                    }else{
                        ct += '<a class="list-group-item typeahead tt-selectable" href="/search?keyword='+data[i]+'">'+data[i]+'</a>'
                    }
                }
                $('.search_nav_input:visible').first().nextAll('.tt-menu:first').find('.tt-dataset-product-title:first').html(ct)
            },
            error: function() {
                alert('Lỗi! vui lòng liên hệ DDshop để được hỗ trợ')
            }
        });
    }
