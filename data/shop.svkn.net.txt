<script type="text/javascript">
  jQuery(document).ready(function($) {
    $('.responsive-menu-controller').click(function(event) {
      if($(this).hasClass('openq')){
        $(this).removeClass('openq');
        $('#mega-menu-container').removeClass('openq');
        $('.close_megamenu').removeClass('openq');
        $('.background_close_megamenu').removeClass('openq');
      }else{
        $(this).addClass('openq');
        $('#mega-menu-container').addClass('openq');
        $('.close_megamenu').addClass('openq');
        $('.background_close_megamenu').addClass('openq');
      }
    });

    $('.close_megamenu').click(function(event) {
      if($(this).hasClass('openq')){
        $(this).removeClass('openq');
        $('#mega-menu-container').removeClass('openq');
        $('.responsive-menu-controller').removeClass('openq');
        $('.background_close_megamenu').removeClass('openq');
      }else{
        $(this).addClass('openq');
        $('#mega-menu-container').addClass('openq');
        $('.responsive-menu-controller').addClass('openq');
        $('.background_close_megamenu').addClass('openq');
      }
    });

    $('.background_close_megamenu').click(function(event) {
      if($(this).hasClass('openq')){
        $(this).removeClass('openq');
        $('#mega-menu-container').removeClass('openq');
        $('.responsive-menu-controller').removeClass('openq');
        $('.close_megamenu').removeClass('openq');
      }else{
        $(this).addClass('openq');
        $('#mega-menu-container').addClass('openq');
        $('.responsive-menu-controller').addClass('openq');
        $('.close_megamenu').addClass('openq');
      }
    });
  });
</script>
