	function newsletter(evt){
		var el = $('p#newsletter > input#email');
		var emailRegExp = /^[_a-z0-9-]+(\.[_a-z0-9-]+)*@[a-z0-9-]+(\.[a-z0-9-]+)*(\.([a-z]){2,4})$/;
		if(el.val()==''){el.focus();$('#notice').html('Xin vui lòng nhập địa chỉ email của bạn');return false;}
		if(!emailRegExp.test(el.val())){
			el.focus();
			$('#notice').html('Email không đúng');
		}else{
			$.ajax({
				type: 'POST',
				url : 'http://ckcpack.com.vn/ajax/newsletter.php',
				data: 'email='+el.val(),
				beforeSend: function(){
					$('#notice').html('<img src="http://ckcpack.com.vn/images/loading-icon.gif" />');
				},
				success: function(result){
					$('#notice').html(result);
				}
			});	
		}
		
	}

		var key;
