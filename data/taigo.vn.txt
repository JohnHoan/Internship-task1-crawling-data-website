<script type="text/javascript">
    $(document).ready(function(){
        $('.tab_click:eq(0)').addClass('mau_them');
        $('.tab_click').click(function(){
            $('.tab_click').removeClass('mau_them');
            $(this).addClass('mau_them');   
            var id_danhmuc = $(this).data('danhmuc');
            var id_list = $(this).data('list');
            
            loadData(1,".load_page_"+id_danhmuc,id_danhmuc,id_list); 
            $(this).parents('.wa_dm').find('.tab_click').val(id_list);
        })
    })
    
    function loadData(page,id_tab,id_danhmuc,id_list){
        $.ajax({
            type: "POST",
            url: "paging_ajax/ajax_paging.php",
            data: {page:page,id_danhmuc:id_danhmuc,id_list:id_list},
            success: function(msg)
            {
                $("#loadbody").fadeOut("fast");
                $(id_tab).html(msg);
                
                  $(id_tab+" .pagination li.active").click(function(){
                    var pager = $(this).attr("p");
                    var id_danhmucr = $(this).parents().parents().parents().attr("data-rel");
                    var id_listr = $(this).parents('.wa_dm').find('.tab_click:eq(0)').val();
                    
                    loadData(pager,".load_page_"+id_danhmucr,id_danhmucr,id_listr);
                }); 
            }
        });
    }

	function number_format (number, decimals, dec_point, thousands_sep) {
          number = (number + '').replace(/[^0-9+\-Ee.]/g, '');
          var n = !isFinite(+number) ? 0 : +number,
              prec = !isFinite(+decimals) ? 0 : Math.abs(decimals),
              sep = (typeof thousands_sep === 'undefined') ? ',' : thousands_sep,
              dec = (typeof dec_point === 'undefined') ? '.' : dec_point,
              s = '',
              toFixedFix = function (n, prec) {
                  var k = Math.pow(10, prec);
                  return '' + Math.round(n * k) / k;
              };
          s = (prec ? toFixedFix(n, prec) : '' + Math.round(n)).split('.');
          if (s[0].length > 3) {
              s[0] = s[0].replace(/\B(?=(?:\d{3})+(?!\d))/g, sep);
          }
          if ((s[1] || '').length < prec) {
              s[1] = s[1] || '';
              s[1] += new Array(prec - s[1].length + 1).join('0');
          }
          return s.join(dec);
}
