            $('.add-cart').on('click', function(e) {
                e.preventDefault();
                loadingOverlay();
                var formID = $(this).data('form');
                var $form = $('#'+formID);

                // 個数フォームのチェック
                var $quantity = $form.parent().find('.quantity');
                if ($quantity.val() < 1) {
                    $quantity[0].setCustomValidity('Vui lòng nhập nhiều hơn 1.');
                    $('.bg-load-overlay').remove();
                    return true;
                } else {
                    $quantity[0].setCustomValidity('');
                }
                e.preventDefault();
                $.ajax({
                    url: $form.attr('action'),
                    type: $form.attr('method'),
                    data: $form.serialize(),
                    dataType: 'json',
                    beforeSend: function(xhr, settings) {
                        // Buttonを無効にする
                        $('.add-cart').prop('disabled', true);
                    }
                }).done(function(data) {
                    if(data.done == false){
                        var message_text = '';
                        $.each(data.messages, function( index, value ) {
                          message_text += value;
                        });
                        $('.bg-load-overlay').remove();

                        swal({
                          icon: "error",
                          text: message_text,
                          title: 'Thật bất tiện'
                        });
                    } else {
                        var totalCartQuantity = parseInt($('.totalCartQuantity').text()) + parseInt($form.find('input[name=quantity]').val())
                        $('.totalCartQuantity').text(totalCartQuantity);
                        $.ajax({
                            url: "https://bephathaopy.com/block/cart",
                            type: 'GET',
                            dataType: 'html'
                        }).done(function(html) {
                            showCartModal(html, '');
                            $('.count-cart').find('span').html(totalCartQuantity);
                        }).always(function(data) {
                            $('.bg-load-overlay').remove();
                        });
                    }
                }).fail(function(data) {
                    swal({
                          icon: "error",
                          text: 'Không thể thêm sản phẩm vào giỏ hàng. Vui lòng chọn các điều kiện trước khi thêm.',
                          title: 'Thật bất tiện'
                        });
                }).always(function(data) {
                    // Buttonを有効にする
                    $('.add-cart').prop('disabled', false);

                });
            });
        });
