    idInnerHtml = '#id-inner-html',
    idWrapArticle = '#id-wrap-article',
    actionType,
    show, 
    itemTemplateTpl,
    wrapTemplateTpl, 
    runSlide
}) {

        type: "article",
        show: 12

    var target = `${idInnerHtml}`;

            $(`${idWrapArticle}`).hide()
        }
        Hura.Template.render(target, listItem);
        if(runSlide){
            runMultipleSwiper(runSlide)
        }


    })

}
